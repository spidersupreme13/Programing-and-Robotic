#include <Servo.h>

Servo miServo;
int sensorPin = 2;
bool estado = false;         // Estado actual del servo
bool sensorAnterior = HIGH;  // Para detectar cambio

void setup() {
    miServo.attach(9);
    pinMode(sensorPin, INPUT);
    Serial.begin(9600);
}

void loop() {
    bool lecturaSensor = digitalRead(sensorPin);
    // Detecta flanco de bajada (de 1 a 0)
    if (sensorAnterior == HIGH && lecturaSensor == LOW) {
        estado = !estado;  // Cambia el estado (true <-> false)
        if (estado) {
            miServo.write(90);
            Serial.println("Servo a 90°");
        } else {
            miServo.write(0);
            Serial.println("Servo a 0°");
        }
        delay(300); // Pequeño delay para evitar rebotes
    }
    sensorAnterior = lecturaSensor;
}
